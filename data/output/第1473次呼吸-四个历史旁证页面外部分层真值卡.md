## 验证命令
```sh
bash -lc '
set -euo pipefail
OUT="/Users/a333/Desktop/问路唯一开发的文件夹/问路的弟弟/wenLuDemo/data/output/第1473次呼吸-四个历史旁证页面外部分层真值卡.md"
TMPDIR="$(mktemp -d)"
trap "rm -rf \"$TMPDIR\"" EXIT
curl --noproxy "*" -L -s -o "$TMPDIR/wechat.html" -w "%{http_code}" https://web.wechat.com/ > "$TMPDIR/wechat.code"
curl --noproxy "*" -L -s -o "$TMPDIR/okjike.html" -w "%{http_code}" https://web.okjike.com/publish > "$TMPDIR/okjike.code"
curl --noproxy "*" -L -s -o "$TMPDIR/weibo.html" -w "%{http_code}" https://weibo.com/ > "$TMPDIR/weibo.code"
curl --noproxy "*" -L -s -o "$TMPDIR/xhs.html" -w "%{http_code}" https://www.xiaohongshu.com/publish/publish > "$TMPDIR/xhs.code"
grep -q "`https://web.wechat.com/` | `$(cat "$TMPDIR/wechat.code")` | 历史可达旁证" "$OUT"
grep -q "`https://web.okjike.com/publish` | `$(cat "$TMPDIR/okjike.code")` | 历史可达旁证" "$OUT"
grep -q "`https://weibo.com/` | `$(cat "$TMPDIR/weibo.code")` | 历史可达旁证" "$OUT"
grep -q "`https://www.xiaohongshu.com/publish/publish` | `$(cat "$TMPDIR/xhs.code")` | 历史失败旁证" "$OUT"
grep -qi "visitor\|passport" "$TMPDIR/weibo.html"
grep -qi "404\|页面不见了\|error" "$TMPDIR/xhs.html"
'
```
